1787 template<typename Block, typename Allocator>
1788 constexpr dynamic_bitset<Block, Allocator>::dynamic_bitset(size_type nbits,
1789  unsigned long long init_val,
1790  const allocator_type& allocator)
1791  : m_blocks(blocks_required(nbits), allocator), m_bits_number(nbits)
1792 {
1793  if(nbits == 0 || init_val == 0)
1794  {
1795  return;
1796  }
1797 
1798  constexpr size_type init_val_required_blocks = sizeof(unsigned long long) / sizeof(block_type);
1799  if constexpr(init_val_required_blocks == 1)
1800  {
1801  m_blocks[0] = init_val;
1802  }
1803  else
1804  {
1805  const unsigned long long block_mask = static_cast<unsigned long long>(one_block);
1806  const size_type blocks_to_init = std::min(m_blocks.size(), init_val_required_blocks);
1807  for(size_type i = 0; i < blocks_to_init; ++i)
1808  {
1809  m_blocks[i] = block_type((init_val >> (i * bits_per_block) & block_mask));
1810  }
1811  }
1812  sanitize();
1813 }
